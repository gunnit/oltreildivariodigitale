Esempi di Codice

Python 
# Esempio di automazione per analisi vendite
import pandas as pd
from sklearn.linear_model import LinearRegression

# Carica dati vendite
data = pd.read_csv('vendite_2024.csv')

# Crea modello predittivo semplice
model = LinearRegression()
X = data[['mese', 'marketing_budget', 'stagionalita']]
y = data['vendite']

model.fit(X, y)

# Predizione vendite prossimo trimestre
next_quarter = [[10, 15000, 0.8], [11, 18000, 1.2], [12, 25000, 1.5]]
predictions = model.predict(next_quarter)

print(f"Vendite previste: {predictions}")
